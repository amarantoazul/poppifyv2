-- Esquema para la tabla 'pedidos'
-- Apto para el editor SQL de Supabase

-- 1. Crear el tipo ENUM para el estatus
CREATE TYPE estatus_pedido AS ENUM ('Pendiente', 'En Proceso', 'Entregado', 'Cancelado');

-- 2. Crear la tabla 'pedidos'
CREATE TABLE pedidos (
    id TEXT PRIMARY KEY,
    folio TEXT NOT NULL,
    cliente TEXT NOT NULL,
    f_compra DATE NOT NULL,
    f_entrega DATE NOT NULL,
    estatus estatus_pedido NOT NULL,
    turno TEXT NOT NULL,
    sucursal TEXT NOT NULL,
    created_at TIMESTAMPTZ DEFAULT now()
);
