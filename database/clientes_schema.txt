-- Esquema para la tabla 'clientes'
-- Apto para el editor SQL de Supabase

-- 1. Crear el tipo ENUM para el estatus
CREATE TYPE estatus_cliente AS ENUM ('Activo', 'Inactivo', 'Potencial');

-- 2. Crear la tabla 'clientes'
CREATE TABLE clientes (
    id TEXT PRIMARY KEY,
    folio TEXT NOT NULL,
    estatus estatus_cliente NOT NULL,
    f_entrega DATE NOT NULL,
    cliente TEXT NOT NULL,
    correo TEXT NOT NULL,
    telefono TEXT NOT NULL,
    destinatario TEXT NOT NULL,
    tel_destino TEXT NOT NULL,
    created_at TIMESTAMPTZ DEFAULT now()
);
