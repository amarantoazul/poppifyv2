-- Esquema para la tabla 'personal'
-- Apto para el editor SQL de Supabase

-- 1. Crear el tipo ENUM para el estatus
CREATE TYPE estatus_personal AS ENUM ('Pendiente', 'En Diseño', 'Listo para Entrega', 'Entregado');

-- 2. Crear la tabla 'personal'
CREATE TABLE personal (
    id TEXT PRIMARY KEY,
    folio TEXT NOT NULL,
    estatus estatus_personal NOT NULL,
    f_entrega DATE NOT NULL,
    cliente TEXT NOT NULL,
    dedicatoria TEXT,
    notas TEXT,
    created_at TIMESTAMPTZ DEFAULT now()
);
